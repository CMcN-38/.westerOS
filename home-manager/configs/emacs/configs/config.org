    (defun cmcn/--nixos-rebuild-sentinel (buf status)
      "One-shot hook: reload config on successful rebuild, then remove itself."
      (remove-hook 'compilation-finish-functions #'cmcn/--nixos-rebuild-sentinel)
      (if (string= status "finished\n")
          (progn
            (message "Rebuild complete, reloading config...")
            (when (file-exists-p "~/.config/emacs/config.el")
              (delete-file "~/.config/emacs/config.el"))
            (load-file "~/.config/emacs/init.el"))
        (message "Rebuild failed — see *compilation* buffer")))

    (defun cmcn/nixos-rebuild-and-reload ()
      "Run nixos-rebuild switch (NixOS module), then reload emacs config.

* IVY (COUNSEL)
+ Ivy, a generic completion mechanism for Emacs.
+ Counsel, a collection of Ivy-enhanced versions of common Emacs commands.
+ Ivy-rich allows us to add descriptions alongside the commands in M-x.

#+begin_src emacs-lisp

    :diminish
    :after ivy
    :config (counsel-mode))

  (use-package ivy :demand t :ensure t
    :diminish
    :bind
    (("C-c C-r" . ivy-resume)
     ("C-x B" . ivy-switch-buffer-other-window))
    :custom
    (setq ivy-use-virtual-buffers t)
    (setq ivy-count-format "(%d/%d) ")
    (setq enable-recursive-minibuffers t)
    :config
    (ivy-mode))
